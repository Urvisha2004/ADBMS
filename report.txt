A cursor is used to fetch and process multiple rows one by one from a table so that we can format and display a report properly.

➡ Allows row-by-row processing, which is required for:
✔ Reports
✔ Calculations
✔ Conditional printing
✔ Proper headings & formatting

Step 1: Create Table
CREATE TABLE employee (
    emp_id NUMBER PRIMARY KEY,
    emp_name VARCHAR2(50),
    department VARCHAR2(30),
    salary NUMBER(8,2)
);


Step 2: Insert Sample Records
INSERT INTO employee VALUES (101, 'Rahul', 'HR', 25000);
INSERT INTO employee VALUES (102, 'Amit', 'IT', 40000);
INSERT INTO employee VALUES (103, 'Neha', 'Finance', 35000);
INSERT INTO employee VALUES (104, 'Priya', 'IT', 45000);

COMMIT;


Step 3: Report Generation Using SQL
SELECT emp_id, emp_name, department, salary
FROM employee
ORDER BY department;


Step 4: Report Generation Using PL/SQL (Practical Program)
SET SERVEROUTPUT ON;

DECLARE
    CURSOR emp_cursor IS
        SELECT emp_id, emp_name, department, salary FROM employee;
        
    emp_record emp_cursor%ROWTYPE;
BEGIN
    DBMS_OUTPUT.PUT_LINE('EMP_ID   NAME     DEPARTMENT   SALARY');
    DBMS_OUTPUT.PUT_LINE('------------------------------------');

    OPEN emp_cursor;
    LOOP
        FETCH emp_cursor INTO emp_record;
        EXIT WHEN emp_cursor%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE(
            emp_record.emp_id || '   ' ||
            emp_record.emp_name || '   ' ||
            emp_record.department || '   ' ||
            emp_record.salary
        );
    END LOOP;

    CLOSE emp_cursor;
END;
/
